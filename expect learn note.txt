#!/usr/bin/expect -f
set ipaddr "localhost"
set passwd "iforgot"
spawn ssh root@$ipaddr              #spawn   意思是执行命令，expect内命令，shell中不存在
expect {
"yes/no" { send "yes\r"; exp_continue} #exp_continue可以继续执行下面的匹配
"password:" { send "$passwd\r" }
}
expect "]# "
send "touch a.txt\r"                       #意思为发送命令
send "exit\r"
expect eof
exit
